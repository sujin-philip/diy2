267 void
268 diy::RegularDecomposer<Bounds>::
269 decompose(int rank, const Creator& create)
270 {
271  std::vector<int> gids;
272  assigner.local_gids(rank, gids);
273  for (int i = 0; i < (int)gids.size(); ++i)
274  {
275  int gid = gids[i];
276 
277  DivisionsVector coords;
278  gid_to_coords(gid, coords);
279 
280  Bounds core, bounds;
281  fill_bounds(core, coords);
282  fill_bounds(bounds, coords, true);
283 
284  // Fill link with all the neighbors
285  Link link(dim, core, bounds);
286  std::vector<int> offsets(dim, -1);
287  offsets[0] = -2;
288  while (!all(offsets, 1))
289  {
290  // next offset
291  int i;
292  for (i = 0; i < dim; ++i)
293  if (offsets[i] == 1)
294  offsets[i] = -1;
295  else
296  break;
297  ++offsets[i];
298 
299  if (all(offsets, 0)) continue; // skip ourselves
300 
301  DivisionsVector nhbr_coords(dim);
302  int dir = 0;
303  bool inbounds = true;
304  for (int i = 0; i < dim; ++i)
305  {
306  nhbr_coords[i] = coords[i] + offsets[i];
307 
308  // wrap
309  if (nhbr_coords[i] < 0)
310  {
311  if (wrap[i])
312  {
313  nhbr_coords[i] = divisions[i] - 1;
314  link.add_wrap(Direction(1 << 2*i));
315  }
316  else
317  inbounds = false;
318  }
319 
320  if (nhbr_coords[i] >= divisions[i])
321  {
322  if (wrap[i])
323  {
324  nhbr_coords[i] = 0;
325  link.add_wrap(Direction(1 << (2*i + 1)));
326  }
327  else
328  inbounds = false;
329  }
330 
331  // NB: this needs to match the addressing scheme in dir_t (in constants.h)
332  if (offsets[i] == -1)
333  dir |= 1 << (2*i + 0);
334  if (offsets[i] == 1)
335  dir |= 1 << (2*i + 1);
336  }
337  if (!inbounds) continue;
338 
339  int nhbr_gid = coords_to_gid(nhbr_coords);
340  BlockID bid; bid.gid = nhbr_gid; bid.proc = assigner.rank(nhbr_gid);
341  link.add_neighbor(bid);
342 
343  Bounds nhbr_bounds;
344  fill_bounds(nhbr_bounds, nhbr_coords);
345  link.add_bounds(nhbr_bounds);
346 
347  link.add_direction(static_cast<Direction>(dir));
348  }
349 
350  create(gid, core, bounds, domain, link);
351  }

450 void
451 diy::RegularDecomposer<Bounds>::
452 fill_divisions(int dim, int nblocks, std::vector<int>& divisions)
453 {
454  int prod = 1; int c = 0;
455  for (unsigned i = 0; i < dim; ++i)
456  if (divisions[i] != 0)
457  {
458  prod *= divisions[i];
459  ++c;
460  }
461 
462  if (nblocks % prod != 0)
463  {
464  std::cerr << "Incompatible requirements" << std::endl;
465  return;
466  }
467 
468  if (c == divisions.size())
469  return;
470 
471  std::vector<unsigned> factors;
472  factor(factors, nblocks/prod);
473 
474  // Fill the missing divs using LPT algorithm
475  std::vector<unsigned> missing_divs(divisions.size() - c, 1);
476  for (int i = factors.size() - 1; i >= 0; --i)
477  *std::min_element(missing_divs.begin(), missing_divs.end()) *= factors[i];
478 
479  c = 0;
480  for (unsigned i = 0; i < dim; ++i)
481  if (divisions[i] == 0)
482  divisions[i] = missing_divs[c++];
483 }
